js/app.js
'use strict'; var Marionette = require('./libs/marionette'), Backbone = require('./libs/backbone'), app;

of type Marionette

Create the application instance

app = new Marionette.Application({ regions: { mainRegion: '#app' } });

Here is where all app configuration should go Like app level event channels or app handlers

Create the main router for the application

app.router = new Backbone.Router();

Create handler for setting page title

app.commands.setHandler('setTitle', function(title) { document.title = document.title + ' - ' + title; });

Create handler for updating url, calling controller method if specified

app.commands.setHandler('navigate', function(opts) { app.router.navigate(opts.path, opts.trigger || false) });

Once the app is started all routers must be instantiated, start Backbone.history

app.on('start', function() { if (Backbone.history) { Backbone.history.start(); } }); module.exports = app;
